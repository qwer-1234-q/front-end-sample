    feature[i].addEventListener('change', () => {
        if (feature[i].checked == true) {
            tmp += 1;
        }
    });
    n_feature = tmp;

    if (n_feature == 4) {
        all_no_select = true;
        infoButton.innerText = "Deselect All";

        infoButton.innerText = "Select All";
    }
    rander();
};
